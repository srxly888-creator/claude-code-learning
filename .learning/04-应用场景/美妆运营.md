### **核心代码示例**

#### **1. 内容生成器**
```python
# src/generator.py

from claude_code import Client
import yaml

class ContentGenerator:
    def __init__(self, config_path="config/brand.yaml"):
        with open(config_path) as f:
            self.config = yaml.safe_load(f)
        self.client = Client()

    def generate_copywriting(
        self,
        product: str,
        platform: str,
        tone: str = "professional",
        length: int = 300
    ):
        """生成产品文案"""
        prompt = f"""
        为美妆品牌'{self.config['brand_name']}'的新品{product}创作社媒文案。

        要求：
        - 平台：{platform}
        - 风格：{tone}
        - 长度：{length}字
        - 品牌调性：{self.config['brand_tone']}
        - 核心卖点：{', '.join(self.config['key_selling_points'])}
        """

        result = self.client.generate(
            prompt=prompt,
            model="claude-3-opus-20240229",
            max_tokens=1000
        )

        return result.content

    def generate_video_script(
        self,
        topic: str,
        duration: int = 60,
        style: str = "tutorial"
    ):
        """生成视频脚本"""
        prompt = f"""
        创作抖音美妆教程视频脚本。

        主题：{topic}
        时长：{duration}秒
        风格：{style}
        品牌调性：{self.config['brand_tone']}

        输出格式：
        - 场景描述
        - 旁白/对话
        - 镜头切换
        - 时间节点
        """

        result = self.client.generate(prompt=prompt)
        return result.content
```

#### **2. 分析器**
```python
# src/analyzer.py

from claude_code import Client
import pandas as pd

class SocialMediaAnalyzer:
    def __init__(self):
        self.client = Client()

    def analyze_sentiment(self, comments_file: str):
        """分析评论情感"""
        df = pd.read_csv(comments_file)

        results = []
        for comment in df['comment']:
            analysis = self.client.analyze(
                text=comment,
                type="sentiment",
                aspects=["quality", "price", "packaging"]
            )
            results.append(analysis)

        return pd.DataFrame(results)

    def extract_trends(self, data_file: str, time_range: str = "7d"):
        """提取热门趋势"""
        df = pd.read_csv(data_file)

        prompt = f"""
        分析以下美妆社媒数据，提取最近{time_range}的热门趋势。

        数据：
        {df.to_string()}

        输出：
        1. 热门话题（Top 10）
        2. 情感倾向
        3. 用户痛点
        4. 营销机会
        """

        result = self.client.generate(prompt=prompt)
        return result.content
```
